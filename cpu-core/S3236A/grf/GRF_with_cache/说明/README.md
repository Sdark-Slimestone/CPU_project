
## 1.单周期处理器，同一周期堆同一个位置进行读写，这是war或者raw，会被发射逻辑解决

## 2.流水线处理器，假设5级 fi di ex（假设在ex读寄存器） mem wb（写回寄存器），假设指令0对x地址写入（同时写入cache），1随便吧，因为错开了不影响，2读x地址（且不命中需要回填），你可能认为这样会冲突，但实际上指令2读地址时，wb还没有完成，所以指令2实际上读的是流水线的转发旁路，根本不访问寄存器系统，所以压根不读窗口或者全局堆，所以也不触发回填

## 3.同一周期写入同一地址（译码器避免）

## 4.同一周期读同一体（译码器避免）：如果全部命中cache，正常读出；如果cache有不命中，判断不命中的请求是否超出窗口堆带宽（总请求数量，单体数量负载），如果超出就不发射后续指令

//call/ret按自定义isa的语义约定，本身的写回永远都在重叠区域内，所以直接写入cache
if(流水线计数器!=0){ //在fi di rr ex mem wb的流水级下。 流水线计数器从call/ret指令的rr阶段的边界重置为2，往后每周期递减，为0后不变
  if(写入地址块在cache中 && 该地址有效 ){/*写重叠窗口 || 写全局寄存器*/
    Bool 写入cache = 1;
  }
  else if (写入地址块在cache中 && !该地址有效){
    if(本周期该块要cache到Wbuffer   ){ /*本周期有指令要读取这个块，所以触发替换*/
      Bool 写入Wbuffer = 1; //且优先级大于cache到Wbuffer
    }
    else if(!本周期该块cache到Wbuffer ){ /*本周期无指令读取这个块，所以触发替换*/
      Bool 写入cache = 1
    }
    
  }
  else if (!写入地址块在cache中 && !该地址有效){
    if(本周期cache要把要写入的块搬到GRF  ){ /*本周期要写回*/
      写入 GRF;  //且优先级大于Wbuffer到GRF
    }
    if(!本周期cache要把要写入的块搬到GRF ) { /*本周期不写回*/
      写入Wbuffer;
    }
  }
}
